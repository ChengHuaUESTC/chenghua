Go 超时的应用

Mon 11 January 2016
By vantigege

单个gorouting超时

package main

import (
    "fmt"
    "time"
)

func main() {
    ch := make(chan string, 1)
    go func() {
        time.Sleep(2 * time.Second)
        ch <- "Done"
    }()

    select {
    case <-time.After(1 * time.Second):
        fmt.Println("overtime")
    case str := <-ch:
        fmt.Println(str)
    }
}
